```python
"""i18n support for vvrite."""

import importlib

from Foundation import NSLocale

SUPPORTED_LANGUAGES = [
    ("en", "English"),
    ("ko", "한국어"),
    ("ja", "日本語"),
    ("zh-Hans", "简体中文"),
    ("zh-Hant", "繁體中文"),
    ("es", "Español"),
    ("fr", "Français"),
    ("de", "Deutsch"),
    ("pt", "Português"),
    ("ru", "Русский"),
    ("ar", "العربية"),
    ("hi", "हिन्दी"),
    ("tr", "Türkçe"),
    ("it", "Italiano"),
]

_SUPPORTED_CODES = {code for code, _ in SUPPORTED_LANGUAGES}

ASR_LANGUAGE_MAP = {
    "en": "English",
    "ko": "Korean",
    "ja": "Japanese",
    "zh-Hans": "Chinese",
    "zh-Hant": "Chinese",
    "es": "Spanish",
    "fr": "French",
    "de": "German",
    "pt": "Portuguese",
    "ru": "Russian",
    "ar": "Arabic",
    "hi": "Hindi",
    "tr": "Turkish",
    "it": "Italian",
}

_current_locale = "en"
_current_strings = None
_en_strings = None


def _load_strings(locale_code: str) -> dict:
    module_name = locale_code.replace("-", "_")
    try:
        mod = importlib.import_module(f"vvrite.locales.{module_name}")
        return mod.strings
    except (ImportError, AttributeError):
        return {}


def _get_en_strings() -> dict:
    global _en_strings
    if _en_strings is None:
        _en_strings = _load_strings("en")
    return _en_strings


def set_locale(code: str):
    global _current_locale, _current_strings
    _current_locale = code
    if code == "en":
        _current_strings = _get_en_strings()
    else:
        _current_strings = _load_strings(code)


def get_locale() -> str:
    return _current_locale


def resolve_system_locale() -> str:
    try:
        preferred = NSLocale.preferredLanguages()
        if preferred and len(preferred) > 0:
            lang = str(preferred[0])
            # Try exact match first (e.g. "zh-Hans")
            if lang in _SUPPORTED_CODES:
                return lang
            # Try base language (e.g. "en-US" -> "en")
            base = lang.split("-")[0]
            if base in _SUPPORTED_CODES:
                return base
    except Exception:
        pass
    return "en"


def t(key: str, **kwargs) -> str:
    parts = key.split(".")
    # Try current locale
    val = _lookup(parts, _current_strings or _get_en_strings())
    # Fall back to English
    if val is None and _current_locale != "en":
        val = _lookup(parts, _get_en_strings())
    if val is None:
        return key
    if kwargs:
        try:
            return val.format(**kwargs)
        except (KeyError, IndexError):
            return val
    return val


def _lookup(parts: list, strings: dict):
    current = strings
    for part in parts:
        if not isinstance(current, dict):
            return None
        current = current.get(part)
        if current is None:
            return None
    if isinstance(current, str):
        return current
    return None
```

```python
def transcribe(raw_wav_path: str, prefs: Preferences = None) -> str:
    if prefs is None:
        prefs = Preferences()

    from vvrite.locales import ASR_LANGUAGE_MAP

    normalized_path = audio_utils.normalize(raw_wav_path)
    try:
        kwargs = {"max_tokens": prefs.max_tokens}
        custom_words = prefs.custom_words.strip()
        if custom_words:
            kwargs["system_prompt"] = f"Use the following spellings: {custom_words}"

        asr_lang = prefs.asr_language
        if asr_lang != "auto":
            language_param = ASR_LANGUAGE_MAP.get(asr_lang)
            if language_param is None:
                print(f"Unknown ASR language code: {asr_lang}, falling back to auto-detect")
            else:
                kwargs["language"] = language_param

        result = _model.generate(
            normalized_path,
            **kwargs,
        )
        return result.text.strip()
    finally:
        for path in (raw_wav_path, normalized_path):
            try:
                os.unlink(path)
            except OSError:
                pass
```

```python
@objc.typedSelector(b"v@:@")
def uiLanguageChanged_(self, sender):
    index = sender.indexOfSelectedItem()
    if index == 0:
        self._prefs.ui_language = None
    else:
        code = SUPPORTED_LANGUAGES[index - 1][0]
        self._prefs.ui_language = code

    # Show restart dialog
    alert = NSAlert.alloc().init()
    alert.setMessageText_(t("settings.language.restart_message"))
    alert.addButtonWithTitle_(t("settings.language.restart_now"))
    alert.addButtonWithTitle_(t("common.later"))
    response = alert.runModal()

    # Close this window and invalidate cached settings window
    self._window.close()
    from AppKit import NSApp
    delegate = NSApp.delegate()
    if hasattr(delegate, 'invalidateSettingsWindow'):
        delegate.invalidateSettingsWindow()

    if response == NSAlertFirstButtonReturn:
        # Restart the app — use `open` with bundle path, not sys.executable
        # (sys.executable is the PyInstaller bootloader in bundled .app)
        import subprocess
        from Foundation import NSBundle
        bundle = NSBundle.mainBundle().bundlePath()
        subprocess.Popen(["/usr/bin/open", "-n", bundle])
        NSApp.terminate_(None)
```
